type var = string
type i = Mpqf.t
type annot =
| Int
| Real
type binop =
| ADD
| SUB
| MUL
| DIV
| POW
type cmpop =
| EQ
| LEQ
| GEQ
| NEQ
| GT
| LT
type expr =
| Funcall of var * expr list
| Neg of expr
| Binary of binop * expr * expr
| Var of var
| Cst of i
type comparison = expr * cmpop * expr
type 'a boolean =
| Cmp of 'a
| And of 'a boolean * 'a boolean
| Or of 'a boolean * 'a boolean
| Not of 'a boolean
type bexpr = comparison boolean
type dom =
| Finite of i * i

a;b

| Minf of i

-oo; a

| Inf of i

a; +oo

| Set of i list

x1; x2; ...; xn

| Top

-oo; +oo

type decl = annot * var * dom
type decls = decl list
type constrs = bexpr list
type instance = i Libabsolute__.Tools.VarMap.t
type info =
| Exact of instance list
| Unfeasible
| Known of (instance * bool) list
type prog = {
init : decls;
constraints : constrs;
objective : expr option;
solutions : info option;
}
